select class, entity.state, entity.country.countryCode from default where class in ["promotionsnapshot", "clearancesnapshot"] and entity.advertdate > 1 and any s in entity.cond.subConds satisfies any v in s.productG satisfies v > 1778 end end order by entity.endDate DESC
